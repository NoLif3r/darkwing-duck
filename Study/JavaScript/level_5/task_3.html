SCRIPT#1. SUMMED_NUMBERS:

var arr = [];

while (true) {
var newElement = prompt('Enter a number: ', '2');
if (newElement === '' || isNaN(newElement) || newElement === null){
break;
}
arr.push(+newElement);
}

var sum = 0;
for (var i = 0; i < arr.length; i++){
sum += arr[i];
}
alert('sum of your numbers: ' + sum);
