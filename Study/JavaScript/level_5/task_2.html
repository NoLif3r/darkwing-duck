function multiplyNumeric(obj){
for (var num in obj){
if (isNumeric(obj[num]) == true){
obj[num] = obj[num]*2;
}
}
}
function isNumeric(n) {
return !isNaN(parseFloat(n)) && isFinite(n)
}
function logObject(obj){
for (var key in obj){
console.log(key + ' = ' + obj[key]);
}
}
