var tasksCompleted = {
'Anna': 9,
'Ivan': 15,
'Faker': 19,
'Elena': 6,
'Anton': 19
};

function getHighscore(obj){
var maxscore = 0;
var name = '';
for (var key in obj){
if (maxscore < obj[key]){
maxscore = obj[key];
name = key;
}
}
return name + ': ' + maxscore;
}

console.log(getHighscore(tasksCompleted));
