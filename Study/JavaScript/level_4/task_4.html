function truncate(str, max_length){
if (str.length > max_length){
return str.substr(0,max_length) + '...';
}
else{
return str;
}
}

var text_1 = prompt('Enter some text:','This is example text, you can left it if you wanna.')
var length = prompt('Enter how many symbols of text you whant to left. \nOther will be cut off.', '15');
alert(truncate(text_1, length));
